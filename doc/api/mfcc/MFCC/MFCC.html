MFCC(int sampleRate, int fftSize, int numFilters, int numCoefs, {bool energy = true, double preEmphasis = 0.97}){

  _numFilters = numFilters;
  _numCoefs = numCoefs;
  if (!energy) {
    _numCoefs += 1;
  }
  _energy = energy;
  _useEmphasis = (preEmphasis != null);

}
